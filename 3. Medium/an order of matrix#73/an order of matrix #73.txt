std::vector<std::vector<int> > orderedMatrix(int a, int b) {
	std::vector<std::vector<int>> vect(a, std::vector<int>(b));
	int num = 1;
	for (int i = 0; i < a; i++) 
		for (int j = 0; j < b; j++)
			vect[i][j] = num++;
	return vect;
}
